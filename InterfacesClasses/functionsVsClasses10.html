      // Constructor template for Square
      //
      //     new Square (int, int, int, ColorString)
      //
      // Interpretation: a square with sides parallel to sides of canvas
      
      class Square implements Shape {
          int x;            // x-position of center
          int y;            // y-position of center
          int l;            // length of one side
          ColorString c;    // color of square
          Image img;        // image of the square
      
          // this constructor initializes the fields
      
          Square (int x, int y, int l, ColorString c) {
              this.x = x;
              this.y = y;
              this.l = l;
              this.c = c;
              this.img = Image.squareImage (l, "solid", c);
          }
      
          // methods
      
          // RETURNS: the weight of this shape
      
          public double weight () {
              return l * l;
          }
      
          // RETURNS: a scene like the given one, but with this
          //     shape painted on it
      
          public Scene addToScene (Scene sc) {
              return Image.placeImage (img, x, y, s);
          }
      }
