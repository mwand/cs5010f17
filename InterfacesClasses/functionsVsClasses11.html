      // Constructor template for Composite
      //
      //     new Composite (Shape, Shape)
      //
      // Interpretation: a composite of front and back shapes
      
      class Composite implements Shape {
          Shape front;      // the shape in front
          Shape back;       // the shape in back
      
          // this constructor initializes the fields
      
          Composite (Shape front, Shape back) {
              this.front = front;
              this.back = back;
          }
      
          // methods
      
          // RETURNS: the weight of this shape
      
          public double weight () {
              return front.weight() + back.weight();
          }
      
          // RETURNS: a scene like the given one, but with this
          //     shape painted on it
      
          public Scene addToScene (Scene sc) {
              return front.addToScene (back.addToScene (sc));
          }
      }
