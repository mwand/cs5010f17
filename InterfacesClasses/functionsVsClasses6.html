A Few of the Help Functions 

      (define (my-circle-weight s)
        (* pi (sqr (my-circle-r s))))
      
      (define (my-square-weight s)
        (sqr (my-square-l s)))
      
      (define (my-composite-weight s)
        (+ (weight (my-composite-front s))    ; note recursion
           (weight (my-composite-back s))))   ;     through weight
      
      (define (my-composite-add-to-scene s scene)
        ;; paint the back image first, then the front
        (let* ((scene2 (add-to-scene (my-composite-front s)))
               (scene3 (add-to-scene (my-composite-back s))))
          scene3))
